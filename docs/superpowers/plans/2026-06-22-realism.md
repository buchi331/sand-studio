```ts
export interface LiquidParams {
  /** How far it searches sideways for a lower resting spot each step. */
  dispersion: number
  /** Per-step probability it is allowed to move sideways at all. */
  flowChance: number
}

export const LIQUID_PARAMS: Partial<Record<MaterialId, LiquidParams>> = {
  [Material.Water]: { dispersion: 6, flowChance: 1.0 },
  [Material.Oil]: { dispersion: 2, flowChance: 0.4 }
}
```

```ts
  private updateLiquid(x: number, y: number, idx: number): void {
    const m = this.cells[idx]
    const below = this.cellAt(x, y + 1)
    if (below !== OOB && this.canSink(m, below)) {
      this.swap(idx, this.index(x, y + 1))
      return
    }
    const dir = this.rng() < 0.5 ? -1 : 1
    for (const dx of [dir, -dir]) {
      const diag = this.cellAt(x + dx, y + 1)
      if (diag !== OOB && this.canSink(m, diag)) {
        this.swap(idx, this.index(x + dx, y + 1))
        return
      }
    }
    // Sideways flow, modulated by per-liquid viscosity.
    const params = LIQUID_PARAMS[m as keyof typeof LIQUID_PARAMS]
    const dispersion = params ? params.dispersion : 4
    if (params && this.rng() >= params.flowChance) return // too viscous this step
    const hdir = this.rng() < 0.5 ? -1 : 1
    if (this.flowSideways(x, y, idx, hdir, dispersion)) return
    this.flowSideways(x, y, idx, -hdir, dispersion)
  }
```

Replace the `flowSideways` method with a version that takes the dispersion:

```ts
  private flowSideways(
    x: number,
    y: number,
    idx: number,
    dir: number,
    dispersion: number
  ): boolean {
    let target = -1
    for (let s = 1; s <= dispersion; s++) {
      const nx = x + dir * s
      if (this.cellAt(nx, y) !== Material.Empty) break
      target = nx
    }
    if (target < 0) return false
    this.swap(idx, this.index(target, y))
    return true
  }
```

```ts
import { Canvas2DRenderer } from './Canvas2DRenderer'
import type { Renderer } from './Renderer'

export type RendererBackend = 'webgl2' | 'canvas2d'

export interface RendererHandle {
  renderer: Renderer
  backend: RendererBackend
}

/** Probe WebGL2 support on a throwaway canvas (never binds the real one). */
export function isWebGL2Available(): boolean {
  try {
    if (typeof document === 'undefined') return false
    const probe = document.createElement('canvas')
    return !!probe.getContext('webgl2')
  } catch {
    return false
  }
}

/**
 * Pick the best available renderer. WebGL2 is wired in Phase ③; until then this
 * always returns the Canvas2D renderer.
 */
export function createRenderer(): RendererHandle {
  return { renderer: new Canvas2DRenderer(), backend: 'canvas2d' }
}
```

Replace the body of the `render` per-cell loop (the `for` loop over `cells`) with this. The key change: stable per-cell grain variation from a coordinate hash, fire colour from `life` (temperature), steam fade from `life`.

```ts
    const cells = grid.cells
    const life = grid.life
    const w = grid.width

    for (let i = 0; i < cells.length; i++) {
      const m = cells[i]
      const color = COLORS[m] ?? COLORS[Material.Empty]
      let r = color[0]
      let g = color[1]
      let b = color[2]

      if (m === Material.Fire) {
        // temperature: hot core (life high) -> white, cooling -> deep red
        const t = Math.min(1, life[i] / FIRE_LIFE)
        r = 255
        g = clamp8(70 + t * 170)
        b = clamp8(t * t * 90)
      } else if (m === Material.Steam) {
        const t = Math.min(1, life[i] / 70)
        const v = clamp8(150 + t * 70)
        r = v
        g = v
        b = clamp8(v + 8)
      } else {
        const amp = VARIATION[m] ?? 0
        if (amp > 0) {
          // stable grain: hash of x,y in [-1,1]
          const x = i % w
          const y = (i / w) | 0
          const n = ((hash(x * 73856093 ^ y * 19349663) & 0xff) / 255) * 2 - 1
          const d = n * amp * 255
          r = clamp8(r + d)
          g = clamp8(g + d)
          b = clamp8(b + d)
        }
      }

      const o = i << 2
      buf[o] = r
      buf[o + 1] = g
      buf[o + 2] = b
      buf[o + 3] = 255
    }

    ctx.putImageData(image, 0, 0)
```

## Task 7: Implement WebGL2Renderer with bloom

**Files:**
- Create: `src/render/WebGL2Renderer.ts`

- [ ] **Step 1: Create the renderer**

Create `src/render/WebGL2Renderer.ts`. It uploads `cells`/`life` to a texture, draws the scene (palette colour + grain + fire temperature + water highlight) to an FBO, extracts emissive pixels, blurs them (2-pass), and composites scene + additive bloom to the screen. The WebGL2 context uses `preserveDrawingBuffer: true` so MediaRecorder capture works (Phase ④).

```ts
import createREGL, { type Regl, type Framebuffer2D, type Texture2D } from 'regl'
import { COLORS, Material, FIRE_LIFE } from '../sim/materials'
import type { GridView, Renderer } from './Renderer'

// 9-entry palette as a 9x1 RGBA texture: rgb = colour, a = unused.
function paletteData(): Uint8Array {
  const data = new Uint8Array(COLORS.length * 4)
  for (let i = 0; i < COLORS.length; i++) {
    data[i * 4] = COLORS[i][0]
    data[i * 4 + 1] = COLORS[i][1]
    data[i * 4 + 2] = COLORS[i][2]
    data[i * 4 + 3] = 255
  }
  return data
}

const QUAD = [-1, -1, 1, -1, -1, 1, 1, 1]

export class WebGL2Renderer implements Renderer {
  private regl: Regl | null = null
  private gridTex: Texture2D | null = null
  private gridBuf: Uint8Array | null = null
  private scene: Framebuffer2D | null = null
  private bright: Framebuffer2D | null = null
  private blurA: Framebuffer2D | null = null
  private blurB: Framebuffer2D | null = null
  private drawScene: ((p: object) => void) | null = null
  private drawBright: ((p: object) => void) | null = null
  private drawBlur: ((p: object) => void) | null = null
  private drawComposite: ((p: object) => void) | null = null
  private palette: Texture2D | null = null
  private gw = 0
  private gh = 0
  private dw = 0
  private dh = 0

  init(canvas: HTMLCanvasElement, width: number, height: number): void {
    const gl = canvas.getContext('webgl2', {
      preserveDrawingBuffer: true, // required so captureStream() records real frames
      alpha: false,
      antialias: false
    })
    if (!gl) throw new Error('WebGL2 unavailable')
    const regl = createREGL({ gl })
    this.regl = regl
    this.gw = width
    this.gh = height
    this.dw = canvas.width || width
    this.dh = canvas.height || height

    this.gridBuf = new Uint8Array(width * height * 4)
    this.gridTex = regl.texture({
      width,
      height,
      format: 'rgba',
      type: 'uint8',
      mag: 'nearest',
      min: 'nearest'
    })
    this.palette = regl.texture({
      width: COLORS.length,
      height: 1,
      format: 'rgba',
      type: 'uint8',
      mag: 'nearest',
      min: 'nearest',
      data: paletteData()
    })

    const fbo = () =>
      regl.framebuffer({
        color: regl.texture({ width: this.dw, height: this.dh, mag: 'linear', min: 'linear' }),
        depth: false
      })
    this.scene = fbo()
    this.bright = fbo()
    this.blurA = fbo()
    this.blurB = fbo()

    const position = regl.buffer(QUAD)
    const base = { attributes: { position }, count: 4, primitive: 'triangle strip' as const }

    this.drawScene = regl({
      ...base,
      framebuffer: regl.prop<{ dst: Framebuffer2D }, 'dst'>('dst'),
      uniforms: {
        uGrid: () => this.gridTex,
        uPalette: () => this.palette,
        uGridSize: () => [this.gw, this.gh],
        uFireLife: FIRE_LIFE,
        uFireId: Material.Fire,
        uWaterId: Material.Water,
        uSteamId: Material.Steam
      },
      vert: `
        precision highp float;
        attribute vec2 position;
        varying vec2 vUv;
        void main() { vUv = position * 0.5 + 0.5; gl_Position = vec4(position, 0, 1); }
      `,
      frag: `
        precision highp float;
        varying vec2 vUv;
        uniform sampler2D uGrid, uPalette;
        uniform vec2 uGridSize;
        uniform float uFireLife, uFireId, uWaterId, uSteamId;
        float hash(vec2 p){ return fract(sin(dot(p, vec2(127.1,311.7))) * 43758.5453); }
        void main() {
          vec2 uv = vec2(vUv.x, 1.0 - vUv.y);          // flip Y: row 0 = top
          vec4 cell = texture2D(uGrid, uv);
          float id = floor(cell.r * 255.0 + 0.5);
          float life = cell.g;                          // already 0..1
          vec3 col = texture2D(uPalette, vec2((id + 0.5) / 9.0, 0.5)).rgb;

          if (abs(id - uFireId) < 0.5) {
            float t = clamp(life * 255.0 / uFireLife, 0.0, 1.0);
            col = vec3(1.0, 0.27 + t * 0.67, t * t * 0.35);
          } else if (abs(id - uSteamId) < 0.5) {
            float t = clamp(life * 255.0 / 70.0, 0.0, 1.0);
            col = vec3(0.6 + t * 0.27);
          } else if (abs(id - uWaterId) < 0.5) {
            vec2 above = uv - vec2(0.0, 1.0 / uGridSize.y);
            float aboveId = floor(texture2D(uGrid, above).r * 255.0 + 0.5);
            if (abs(aboveId - uWaterId) > 0.5) col += 0.12;   // surface highlight
            col *= 0.9 + 0.1 * uv.y;                           // subtle depth shade
          } else if (id > 0.5) {
            vec2 c = floor(uv * uGridSize);
            col += (hash(c) * 2.0 - 1.0) * 0.12;              // grain
          }
          gl_FragColor = vec4(col, 1.0);
        }
      `
    })

    this.drawBright = regl({
      ...base,
      framebuffer: this.bright,
      uniforms: {
        uScene: () => this.scene,
        uGrid: () => this.gridTex,
        uFireId: Material.Fire,
        uSteamId: Material.Steam
      },
      vert: `
        precision highp float; attribute vec2 position; varying vec2 vUv;
        void main(){ vUv = position*0.5+0.5; gl_Position = vec4(position,0,1); }
      `,
      frag: `
        precision highp float; varying vec2 vUv;
        uniform sampler2D uScene, uGrid; uniform float uFireId, uSteamId;
        void main(){
          float id = floor(texture2D(uGrid, vec2(vUv.x, 1.0-vUv.y)).r * 255.0 + 0.5);
          float em = (abs(id-uFireId)<0.5 || abs(id-uSteamId)<0.5) ? 1.0 : 0.0;
          gl_FragColor = vec4(texture2D(uScene, vUv).rgb * em, 1.0);
        }
      `
    })

    this.drawBlur = regl({
      ...base,
      framebuffer: regl.prop<{ dst: Framebuffer2D }, 'dst'>('dst'),
      uniforms: {
        uTex: regl.prop<{ src: Framebuffer2D }, 'src'>('src'),
        uDir: regl.prop<{ dir: number[] }, 'dir'>('dir'),
        uRes: () => [this.dw, this.dh]
      },
      vert: `
        precision highp float; attribute vec2 position; varying vec2 vUv;
        void main(){ vUv = position*0.5+0.5; gl_Position = vec4(position,0,1); }
      `,
      frag: `
        precision highp float; varying vec2 vUv;
        uniform sampler2D uTex; uniform vec2 uDir, uRes;
        void main(){
          vec2 px = uDir / uRes;
          vec3 sum = texture2D(uTex, vUv).rgb * 0.227;
          sum += texture2D(uTex, vUv + px*1.5).rgb * 0.316;
          sum += texture2D(uTex, vUv - px*1.5).rgb * 0.316;
          sum += texture2D(uTex, vUv + px*3.5).rgb * 0.070;
          sum += texture2D(uTex, vUv - px*3.5).rgb * 0.070;
          gl_FragColor = vec4(sum, 1.0);
        }
      `
    })

    this.drawComposite = regl({
      ...base,
      uniforms: {
        uScene: () => this.scene,
        uBloom: () => this.blurB,
        uIntensity: 1.4
      },
      vert: `
        precision highp float; attribute vec2 position; varying vec2 vUv;
        void main(){ vUv = position*0.5+0.5; gl_Position = vec4(position,0,1); }
      `,
      frag: `
        precision highp float; varying vec2 vUv;
        uniform sampler2D uScene, uBloom; uniform float uIntensity;
        void main(){
          vec3 c = texture2D(uScene, vUv).rgb + texture2D(uBloom, vUv).rgb * uIntensity;
          gl_FragColor = vec4(c, 1.0);
        }
      `
    })
  }

  resize(displayWidth: number, displayHeight: number): void {
    const w = Math.max(1, Math.round(displayWidth))
    const h = Math.max(1, Math.round(displayHeight))
    if (w === this.dw && h === this.dh) return
    this.dw = w
    this.dh = h
    this.scene?.resize(w, h)
    this.bright?.resize(w, h)
    this.blurA?.resize(w, h)
    this.blurB?.resize(w, h)
  }

  render(grid: GridView): void {
    const regl = this.regl
    const buf = this.gridBuf
    if (!regl || !buf || !this.gridTex) return

    const cells = grid.cells
    const life = grid.life
    for (let i = 0; i < cells.length; i++) {
      buf[i * 4] = cells[i]
      buf[i * 4 + 1] = life[i]
      buf[i * 4 + 2] = 0
      buf[i * 4 + 3] = 255
    }
    this.gridTex({
      width: grid.width,
      height: grid.height,
      format: 'rgba',
      type: 'uint8',
      mag: 'nearest',
      min: 'nearest',
      data: buf
    })

    this.drawScene!({ dst: this.scene })
    this.drawBright!({})
    this.drawBlur!({ src: this.bright, dst: this.blurA, dir: [1, 0] })
    this.drawBlur!({ src: this.blurA, dst: this.blurB, dir: [0, 1] })
    this.drawComposite!({})
  }

  dispose(): void {
    this.regl?.destroy()
    this.regl = null
  }
}
```
